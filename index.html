const noBtn = document.getElementById('no-btn'); const yesBtn = document.getElementById('yes-btn'); const mainCard = document.getElementById('main-card'); const successMsg = document.getElementById('success-msg'); const foodBtns = document.querySelectorAll('.food-btn'); // Bỏ URL Webhook của bạn vào đây (Xem hướng dẫn lấy URL ở dưới) const DISCORD_WEBHOOK_URL = 'DÁN_URL_WEBHOOK_CỦA_BẠN_VÀO_ĐÂY'; function moveButton(e) { if(e) e.preventDefault(); noBtn.style.position = 'fixed'; const maxX = window.innerWidth - noBtn.offsetWidth; const maxY = window.innerHeight - noBtn.offsetHeight; let randomX, randomY; const safeDistance = window.innerWidth * 0.2; const currentX = parseInt(noBtn.style.left) || window.innerWidth/2; const currentY = parseInt(noBtn.style.top) || window.innerHeight/2; do { randomX = Math.floor(Math.random() * maxX); randomY = Math.floor(Math.random() * maxY); } while ( Math.abs(randomX - currentX) < safeDistance && Math.abs(randomY - currentY) < safeDistance ); noBtn.style.left = randomX + 'px'; noBtn.style.top = randomY + 'px'; } noBtn.addEventListener('mouseover', moveButton); noBtn.addEventListener('touchstart', moveButton, {passive: false}); yesBtn.addEventListener('click', () => { mainCard.style.display = 'none'; successMsg.style.display = 'block'; }); // XỬ LÝ GỬI THÔNG TIN VỀ ĐIỆN THOẠI foodBtns.forEach(btn => { btn.addEventListener('click', (e) => { const monAn = e.target.innerText; // Nội dung tin nhắn sẽ bắn về Discord của bạn const messageData = { content: `🔔 **THÔNG BÁO HẸN HÒ!**\nCô ấy đã đồng ý đi ăn tối với bạn!\nMón ăn được chọn: **${monAn}** 🥰\nChuẩn bị lên đồ thôi THAI NGUYEN ơi!` }; // Gửi ngầm dữ liệu lên Discord fetch(DISCORD_WEBHOOK_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(messageData) }) .then(() => { // Khi gửi thành công, hiển thị thông báo ngọt ngào cho nàng alert(`Tuyệt vời ông mặt trời! Anh đã nhận được lựa chọn "${monAn}" của em rồi nhé. Hẹn gặp công chúa T6 nhe ❤️`); }) .catch(err => { // Phòng hờ lỗi mạng không gửi được, vẫn hiện alert để giữ mạch tinh tế alert(`Tuyệt vời! Em đã chọn: "${monAn}". Hẹn gặp em T6 nhe ❤️`); console.error('Lỗi gửi webhook:', err); }); }); });
